import { useState, useCallback } from 'react';
import { AgentConfig, Message } from '../types';
import { aiService } from '../services/aiService';

export const useAgent = (config: AgentConfig) => {
  const [messages, setMessages] = useState<Message[]>([]);
  const [loading, setLoading] = useState(false);

  const sendMessage = useCallback(async (content: string) => {
    setLoading(true);
    
    const userMessage: Message = {
      id: Date.now().toString(),
      content,
      role: 'user',
      timestamp: new Date()
    };
    
    setMessages(prev => [...prev, userMessage]);

    try {
      const response = await aiService.sendMessage(config.botId, content);
      
      const assistantMessage: Message = {
        id: (Date.now() + 1).toString(),
        content: response.text,
        role: 'assistant',
        timestamp: new Date()
      };
      
      setMessages(prev => [...prev, assistantMessage]);
    } catch (error) {
      console.error('å‘é€æ¶ˆæ¯å¤±è´¥:', error);
    } finally {
      setLoading(false);
    }
  }, [config.botId]);

  return { messages, sendMessage, loading };
};

import { createSlice, createAsyncThunk } from '@reduxjs/toolkit';

export const sendMessage = createAsyncThunk(
  'agent/sendMessage',
  async ({ botId, message, history = [] }, { rejectWithValue }) => {
    try {
      const response = await fetch('/api/ai/chat', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ botId, message, history })
      });
      
      if (!response.ok) {
        throw new Error('å‘é€æ¶ˆæ¯å¤±è´¥');
      }
      
      return await response.json();
    } catch (error) {
      return rejectWithValue(error.message);
    }
  }
);

const agentSlice = createSlice({
  name: 'agent',
  initialState: {
    messages: [],
    loading: false,
    error: null,
    currentAgent: null
  },
  reducers: {
    clearMessages: (state) => {
      state.messages = [];
    },
    setCurrentAgent: (state, action) => {
      state.currentAgent = action.payload;
    }
  },
  extraReducers: (builder) => {
    builder
      .addCase(sendMessage.pending, (state) => {
        state.loading = true;
        state.error = null;
      })
      .addCase(sendMessage.fulfilled, (state, action) => {
        state.loading = false;
        state.messages.push(action.payload);
      })
      .addCase(sendMessage.rejected, (state, action) => {
        state.loading = false;
        state.error = action.payload;
      });
  }
});

export const { clearMessages, setCurrentAgent } = agentSlice.actions;
export default agentSlice.reducer;

// src/composables/useCloudbase.js
import { ref, onMounted } from 'vue';
import cloudbase from '@cloudbase/js-sdk';

const app = ref(null);
const ai = ref(null);
const initialized = ref(false);

export function useCloudbase() {
  onMounted(async () => {
    if (!initialized.value) {
      app.value = cloudbase.init({
        env: import.meta.env.VITE_CLOUDBASE_ENV_ID
      });
      
      await app.value.auth().signInAnonymously();
      ai.value = app.value.ai();
      initialized.value = true;
    }
  });

  return {
    app: app.value,
    ai: ai.value,
    initialized: initialized.value
  };
}
